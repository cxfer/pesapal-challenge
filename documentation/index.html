Getting Started

Installation

# Prerequisites
flutter --version

# Clone repository
git clone <repository-url>

# Navigate to project
cd pesapal_rdbms

# Install dependencies
flutter pub get

# Run the application
flutter run -d chrome

Quick Start Examples

# Create a sample table
CREATE TABLE products (
  id int primary key,
  name string,
  price double,
  in_stock boolean
);

# Insert sample data
INSERT INTO products (id, name, price, in_stock)
VALUES (1, 'Laptop', 999.99, true);

# Query data
SELECT * FROM products;

# Update data
UPDATE products SET price = 899.99 WHERE id = 1;

# Join with another table
CREATE TABLE inventory (product_id int, quantity int);
INSERT INTO inventory VALUES (1, 50);
JOIN products WITH inventory ON products.id = inventory.product_id;
